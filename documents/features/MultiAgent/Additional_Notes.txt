The Skill Loader (skillLoader.ts)
TypeScript
import * as fs from "node:fs/promises";
import * as path from "node:path";

export interface SkillSummary {
  filename: string;
  content: string;
}

const LOG_PREFIX = "[SkillLoader]";

/**
 * Scans the skills directory and returns all .md files as a formatted string
 * ready to be injected directly into Kayley's system prompt.
 */
export async function getActiveSkillsContext(workspaceRoot: string): Promise<string> {
  const skillsDir = path.join(workspaceRoot, "skills");

  try {
    // Check if the skills directory exists, if not, return empty context
    await fs.access(skillsDir);
  } catch {
    console.log(`${LOG_PREFIX} Skills directory not found at ${skillsDir}. Returning empty skill context.`);
    return "No custom skills have been created yet.";
  }

  try {
    const files = await fs.readdir(skillsDir);
    const mdFiles = files.filter(file => file.endsWith(".md"));

    if (mdFiles.length === 0) {
      return "No custom skills have been created yet.";
    }

    let skillContext = `You currently have ${mdFiles.length} custom skills available. Here are your instruction manuals:\n\n`;

    for (const file of mdFiles) {
      const filePath = path.join(skillsDir, file);
      const content = await fs.readFile(filePath, "utf-8");
      
      // Append each skill file's content into the context payload
      skillContext += `--- BEGIN SKILL: ${file} ---\n`;
      skillContext += `${content.trim()}\n`;
      skillContext += `--- END SKILL: ${file} ---\n\n`;
    }

    return skillContext.trim();

  } catch (error) {
    console.error(`${LOG_PREFIX} Failed to load skills:`, error);
    return "Error loading skills. You may need to troubleshoot your skills directory.";
  }
}

1. The Core Tool: child_process.exec

TypeScript
import { exec } from "node:child_process";
import { promisify } from "node:util";

// Promisify exec so we can use modern async/await syntax
const execAsync = promisify(exec);

export interface TerminalResult {
  stdout: string;
  stderr: string;
  exitCode: number;
}

/**
 * Executes a terminal command in a specific directory.
 */
export async function runTerminalCommand(
  command: string, 
  workspaceRoot: string
): Promise<TerminalResult> {
  console.log(`[Terminal] Running: ${command}`);

  try {
    // We pass 'cwd' to ensure the terminal opens strictly in your project folder
    const { stdout, stderr } = await execAsync(command, {
      cwd: workspaceRoot, 
      timeout: 60000, // 60-second timeout so a hanging command doesn't freeze the agent
      // shell: "powershell.exe" // Optional: Force PowerShell on Windows
    });

    return {
      stdout: stdout.trim(),
      stderr: stderr.trim(),
      exitCode: 0,
